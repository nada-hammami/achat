package tn.esprit.com.foyer.entities;

public class TypeChambrePourcentage {

        this.typeChambre = typeChambre;
    }

    public Float getPourcentage() {
        return pourcentage;
    }

    public void setPourcentage(Float pourcentage) {
        this.pourcentage = pourcentage;
    }

}
